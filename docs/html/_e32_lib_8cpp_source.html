152  byte lByte = lowByte(From8BitValue);
153  byte hByte = highByte(From8BitValue);
154  if ( From8BitValue > 0x3FFF ) return -1; // Value is too big for a 14-bit Value
155  hByte = hByte << 1; // Make Room on hByte for bit-7 of lByte
156  bitWrite(hByte,0,bitRead(lByte,7)); // Transfer lByte<7> onto hByte<0>
