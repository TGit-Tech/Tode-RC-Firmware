177 void Sys::RFLoop() {
178  // Receive RF Data
179  if ( RF==0 ) { DBERRORL(("Sys::RFLoop RF==0")) return 0; }
180  if ( RFPCConn!=0 ) { if ( RFPCConn->Value()==1 ) { RFPCConn->Loop(); return;} } // Loop for Radio->PC Connection
181  if ( !RF->PacketAvailable() ) return; // Collect till Available
182 
183  // 1. Check Packet has an assigned Type
184  if ( RF->Packet->Type() == PKT_NOTSET ) {
185  DBERRORL(("Sys::RFLoop PKT_NOTSET"))
186  delete(RF->Packet); RF->Packet = 0; return;
187  }
188 
189  // 2. Obtain TargetTode
190  Tode* TargetTode=0; // Set Packets TargetTode
191  if ( bitRead(RF->Packet->Type(),5) == 1 ) { TargetTode = ThisTode; } // GET/SET = ThisTode
192  else { TargetTode=RFTode(RF->Packet->FromRF(),(RF->Packet->Type()==PKT_GOTCONFIG)); } // GOTS FromRF (GOTCONFIG Create)
193  if ( TargetTode==0 ) { // Check Target Tode
194  DBERRORAL(("Sys::RFLoop TargetTode==0 PacketType:"),(RF->Packet->Type()))
195  delete(RF->Packet); RF->Packet = 0; return;
196  }
197  DBINFOAL(("Sys::RFLoop TargetTode->TodeIndex:"),(TargetTode->TodeIndex))
198 
199  //------------------------- CONFIG ( No version control )----------------------------------------------------------
200  if ( RF->Packet->Type() == PKT_GOTCONFIG ) { DBINFOL(("Sys::RFLoop PKT_GOTCONFIG"))
201  RF->Packet->SaveTodeConfig( TargetTode->EEAddress() ); // Save the Tode Configuration
202  TargetTode->EELoadDevices(); // Reload Tode
203  delete(RF->Packet); RF->Packet = 0; return; // Exit
204  }
205  if ( RF->Packet->Type() == PKT_GETCONFIG ) { DBINFOL(("Sys::RFLoop PKT_GETCONFIG"))
206  TxPacket Pkt(EEPROM.read(EMC_SECNET), PKT_GOTCONFIG,
